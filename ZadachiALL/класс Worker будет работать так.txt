class Worker {
	constructor(name, surname, rate, days) {
		this.name = name;
		this.surname = surname;
		this.rate = rate;
		this.days = days;
	}

	getSalary() {
		return this.rate * this.days;
	}

  getName() {
		return this.name;
	}
	getSurname() {
		return this.surname;
	}
	getRate () {
		return this.rate;
	};
	setRate(newRate) {
		this.rate = newRate;
  }

  getDays(){
		return this.days;
	}
	setDays(newDays) {
		this.days = newDays;
	}
	getSalary(){
		return days * rate;
	}
}
